241int main ( void )
242{
243 /* Initialize all modules */
244 SYS_Initialize ( NULL );
245
246 /* Give Shortprotocol Parameters initial Values */
247 SHORTPROTOCOL_Initialize(&shortProt);
248
249 /* Recieve CAN Frames for 5ms */
250 DELAY_Microseconds(5000);
251
252 /* Main Loop */
253 /* The Maximum Loop Time has to be smaller than 2ms, to catch all Messages*80/
254 /* 15k Frames/s MAX and FIFO Length 32 -> 1/15E3 * 32 = 2ms */
255 while(1){
256
257 /* Read All Messages in the CAN FIFOs */
258 CANCOMM_ReadMessages(message_list, message_list_len,
259 interface_list, interface_list_len);
260
261 /* Interpret the RAW CAN Message Data */
262 SIGNALS_Interpret(signal_list, signal_list_len,
263 message_list, message_list_len);
264
265 /* Generate the Commands to be sent to the Display */
266 COMMAND_Generate(signal_list, signal_list_len,
267 &current_command_signal, &shortProt);
268
269 /* Send Messages to Display */
270 SHORTPROTOCOL_Update(&shortProt);
271
272 /* Wait for constant Loop time */
273 DELAY_Microseconds(200);
274
275 /* (Window WDG Reset)*/
276
277
278 }
279
280 /* Execution should not come here during normal operation */
281}
