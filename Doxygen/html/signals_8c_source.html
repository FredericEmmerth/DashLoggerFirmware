16void SIGNALS_Interpret(signals_signal* signal_list, uint32_t signal_list_len,
17 cancomm_message* message_list, uint32_t message_list_len){
18
19 uint8_t databuff[CANCOMM_MAXIMUM_DATA_LENGTH];
20 for(uint32_t i=0; i<signal_list_len; i++){
21 if(signal_list[i].type == SIGNALS_CAN_MESSAGE){
22 if(signals_find_data(signal_list[i].id, signal_list[i].interface_number,
23 message_list, message_list_len, databuff) == SIGNALS_FOUND)
24 {
25 switch(signal_list[i].data_type){
26 case SIGNALS_FLOAT_SIGNAL:
27 signal_list[i].value_float =
28 signal_list[i].can_convert_float(databuff);
29 break;
30
31 case SIGNALS_UINT32_T_SIGNAL:
32 signal_list[i].value_uint32_t =
33 signal_list[i].can_convert_uint32_t(databuff);
34 break;
35
36 case SIGNALS_STRING_SIGNAL:
37 signal_list[i].can_convert_string(databuff,
38 &(signal_list[i].value_string));
39 break;
40
41 default:
42 break;
43 }
44 }
45 }else if(signal_list[i].type == SIGNALS_INTERNAL_SIGNAL){
46 switch(signal_list[i].data_type){
47 case SIGNALS_FLOAT_SIGNAL:
48 signal_list[i].value_float =
49 signal_list[i].internal_convert_float(
50 signal_list,signal_list_len);
51 break;
52
53 case SIGNALS_UINT32_T_SIGNAL:
54 signal_list[i].value_uint32_t =
55 signal_list[i].internal_convert_uint32_t(
56 signal_list, signal_list_len);
57 break;
58
59 case SIGNALS_STRING_SIGNAL:
60 signal_list[i].internal_convert_string(signal_list,
61 signal_list_len, &(signal_list[i].value_string));
62 break;
63
64 default:
65 break;
66 }
67 }else if(signal_list[i].type == SIGNALS_DISPLAY_SIGNAL){
68 switch(signal_list[i].data_type){
69
70 case SIGNALS_STRING_SIGNAL:

82signals_result signals_find_data(uint32_t id, uint8_t interface,
83 cancomm_message* message_list, uint32_t message_list_len,
84 uint8_t* data){
85
86 signals_result result = SIGNALS_NOT_FOUND;
87 for(uint32_t i=0; i<message_list_len; i++){
88 if(message_list[i].id == id && message_list[i].interface_number == interface){
89 result = SIGNALS_FOUND;
90 for(uint8_t j=0; j<CANCOMM_MAXIMUM_DATA_LENGTH; j++){
91 data[j] = message_list[i].data[j];
92 }
93 break;
94 }
95 }
96
97 return result;
98}

100signals_result signals_compare_names(uint8_t* first, uint32_t firstlen,
101 uint8_t* second, uint32_t secondlen){
102
103 uint32_t searchlen;
104
105 /* Set searchlen to the shorter length */
106 if(firstlen > secondlen){
107 searchlen = secondlen;
108 }else if (secondlen > firstlen){
109 searchlen = firstlen;
110 }else{
111 /* The Strings have equal length */
112 searchlen = firstlen;
113 }
114
115 signals_result res = SIGNALS_MATCH;
116
117 for(uint32_t i=0; i<searchlen; i++){
118 if(first[i] != second[i]){
119 res = SIGNALS_NO_MATCH;
120 break;
121 }
122 }
123
124 return res;
125}

127signals_signal* signals_find_signal( signals_signal* signal_list,
128 uint32_t signal_list_len, void(*Callback)(void)){
129
130 for(uint32_t i=0; i<signal_list_len; i++){
131 if(Callback == (void(*)(void))signal_list[i].can_convert_float ||
132 Callback == (void(*)(void))signal_list[i].can_convert_uint32_t ||
133 Callback == (void(*)(void))signal_list[i].internal_convert_float ||
134 Callback == (void(*)(void))signal_list[i].internal_convert_uint32_t||
135 Callback == (void(*)(void))signal_list[i].internal_convert_string ||
136 Callback == (void(*)(void))signal_list[i].can_convert_string )
137 {
138 return &(signal_list[i]);
139 }
140 }
141
142 return NULL;
143}

148signals_signal* signals_find_display_signal(signals_signal* signal_list,
149 uint32_t signal_list_len, uint32_t* dispSignalCount, uint32_t needle){
150
151 uint32_t counter = 0;
152 signals_signal* foundNeedle = NULL;
153 for(uint32_t i=0; i<signal_list_len; i++){
154 if(signal_list[i].type == SIGNALS_DISPLAY_SIGNAL){
155 if(needle == counter){
156 foundNeedle = &(signal_list[i]);
157 }
158 counter++;
159 }
160 }
161
162 *dispSignalCount = counter;
163 return foundNeedle;
164}
