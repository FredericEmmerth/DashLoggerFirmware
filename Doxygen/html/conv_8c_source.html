98float CONV_LastLapTime(signals_signal* signal_list, uint32_t signal_list_len){
99
100 static float last_reported_value;
101 static float last_lap;
102
103 float reported_lap_value = signals_find_signal(signal_list, signal_list_len,
104 (void(*)(void))CONV_LapTime)->value_float;
105
106 if(reported_lap_value != last_reported_value){
107 last_lap = last_reported_value;
108 }
109
110
111 last_reported_value = reported_lap_value;
112
113 return last_lap;
114}
