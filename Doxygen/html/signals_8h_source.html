11/**
12 * \defgroup SIGNALS SIGNALS
13 * This Module Handles the Information Interpretation Handling of All Inputs
14 * to Outputs. Signals have a Type, wich changes how it is interpreted and used.
15 * For more on this, refer to \ref signals_data_type and
16 * \ref signals_signal_type. Signals can be chained together. For Example, four
17 * Signals of Type SIGNALS_CAN_MESSAGE could recieve four Temperatures, wich
18 * would then be interpreted to float Values using thier respective Callback
19 * Functions. Another Signal of Type SIGNALS_INTERNAL_SIGNAL could then find the
20 * highest of these Temperatures using its Callback Function. To complete the
21 * Chain, a Signal of Type SIGNALS_DISPLAY_SIGNAL can be used to convert the
22 * float value of the previous Maximising Function to a String Command wich can
23 * be interpreted by a Display.
24 * \addtogroup SIGNALS
25 * \{
26 */
27

107/**
108 * \struct signals_signal_struct
109 *
110 * This Struct defines a Signal. Every Signal should have a friendly Name, to be
111 * easily identified by a human. Every Signal has to have a
112 * \ref signals_signal.type and a \ref signals_signal.data_type. Depending on
113 * the \ref signals_signal.type and \ref signals_signal.data_type other
114 * members of the struct have to be defined. If a signal is of type
115 * SIGNALS_CAN_MESSAGE, the id and the interface_number have to be defined.
116 * The Signal can be either of data_type SIGNALS_FLOAT_SIGNAL,
117 * SIGNALS_UINT32_T_SIGNAL or SIGNALS_STRING_SIGNAL. The corresponding callback
118 * function can_convert_float, can_convert_uint32_t or can_convert_string has
119 * to be defined.
120 *
121 * If a signal is of type SIGNALS_INTERNAL_SIGNAL, it can be of data_type
122 * SIGNALS_FLOAT_SIGNAL, SIGNALS_UINT32_T_SIGNAL or SIGNALS_STRING_SIGNAL. The
123 * corresponding callback function internal_convert_float,
124 * internal_convert_uint32_t or internal_convert_string has to be defined.
125 *
126 * If a signal is of type SIGNALS_DISPLAY_SIGNAL, it's data_type has to be
127 * SIGNALS_STRING_SIGNAL, and its callback function internal_convert_string
128 * has to be defined in a way to produce correct Commands for the Display to
129 * be interpreted. Signals of this Type are read by \ref COMMAND to be sent of
130 * to the Display using the \ref SHORTPROTOCOL.
131 *
132 * A Signal is therefore defined by its type and data_type.
133 *
134 * @todo Combine the Different Signal Types to a Union to use less RAM
135 *
136 */
137struct signals_signal_struct{
138 /** The ID of the CAN Message */
139 uint32_t id;
140
141 /** The CAN Interface Number (unique Identifier for a CAN Interface)*/
142 uint8_t interface_number;
143
144 /** Type of the Signal */
145 signals_signal_type type;
146
147 /** Data Type of the Signal */
148 signals_data_type data_type;
149
150 /** Callback Function for type=SIGNALS_CAN_MESSAGE and
151 * data_type=SIGNALS_FLOAT_SIGNAL */
152 float(*can_convert_float)(uint8_t* data);
153
154 /** Callback Function for type=SIGNALS_INTERNAL_SIGNAL and
155 * data_type=SIGNALS_FLOAT_SIGNAL */
156 float(*internal_convert_float)(signals_signal* signal_list,
157 uint32_t signal_list_len);
158
159 /** The Float Value of this Signal, only used when
160 * data_type=SIGNALS_FLOAT_SIGNAL*/
161 float value_float;
162
163 /** Callback Function for type=SIGNALS_CAN_MESSAGE and
164 * data_type=SIGNALS_UINT32_T_SIGNAL*/
165 uint32_t(*can_convert_uint32_t)(uint8_t* data);
166
167 /** Callback Function for type=SIGNALS_INTERNAL_SIGNAL and
168 * data_type=SIGNALS_UINT32_T_SIGNAL*/
169 uint32_t(*internal_convert_uint32_t)(signals_signal* signal_list,
170 uint32_t signal_list_len);
171
172 /** The uint32_t Value of this Signal, only used when
173 * data_type=SIGNALS_UINT32_T_SIGNAL*/
174 uint32_t value_uint32_t;
175
176 /** Callback Function for type=SIGNALS_CAN_MESSAGE and
177 * data_type=SIGNALS_STRING_SIGNAL*/
178 void (*can_convert_string)(uint8_t* data, SIGNALS_string* string);
179
180 /** Callback Function for type=SIGNALS_INTERNAL_SIGNAL or
181 * type=SIGNALS_DISPLAY_SIGNAL. and data_type SIGNALS_STRING_SIGNAL*/
182 void (*internal_convert_string)(signals_signal* signal_list,
183 uint32_t signal_list_len, SIGNALS_string* string);
184
185 /** The String Value of this Signal, only used when
186 * data_type=SIGNALS_STRING_SIGNAL*/
187 SIGNALS_string value_string;
188
189 /** A friendly Name of the Signal, only used for Debugging */
190 uint8_t friendly_name[SIGNALS_MAXIMUM_NAME_LENGTH];
191
192 /** The Timestamp of the Signal */
193 uint32_t timestamp;
194};
