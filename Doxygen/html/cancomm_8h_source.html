68 /**
69 * @brief Callback Function to Transmit CAN Messages
70 *
71 * This Function matches the Protoype generated by MPLAB Harmony, so this
72 * Function Pointer can be set the the Functions generated by Harmony.
73 *
74 * \param id The ID of the Message
75 * \param length The DLC of the Message
76 * \param data The Data Bytes the Message should have
77 * \param fifoQueueNum The Number of the transmit FIFO
78 * \param mode For non CAN-FD Messages this is CANFD_MODE_NORMAL
79 * \param msgAttr For a Data Frame this is CANFD_MSG_TX_DATA_FRAME
80 *
81 * \return True if the Message was sucessfully added to the transmit FIFO
82 */
83 bool (*MessageTransmit)(uint32_t id,
84 uint8_t length, uint8_t* data,
85 uint8_t fifoQueueNum, CANFD_MODE mode,
86 CANFD_MSG_TX_ATTRIBUTE msgAttr);
87
88 /**
89 * @brief Callback Function to Recieve CAN Messages
90 *
91 * This Function matches the Protoype generated by MPLAB Harmony, so this
92 * Function Pointer can be set the the Functions generated by Harmony.
93 *
94 * \param id The ID of the Recieved Message
95 * \param length The DLC of the Recieved Message
96 * \param data The Data of the Recieved Message
97 * \param timestamp The Timestamp of the Recieved Message
98 * \param fifoNum The FIFO used for recieving Messages
99 * \param msgAttr If the recieved Message was a Data Frame, this is
100 * CANFD_MSG_TX_DATA_FRAME
101 *
102 * \returns False if there is no Message in the Recieve FIFO left
103 *
104 */
105 bool (*MessageReceive)(uint32_t *id,
106 uint8_t *length, uint8_t *data,
107 uint32_t *timestamp, uint8_t fifoNum,
108 CANFD_MSG_RX_ATTRIBUTE *msgAttr);
