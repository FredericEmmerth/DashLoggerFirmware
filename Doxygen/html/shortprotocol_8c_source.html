17SHORTPROTOCOL_status SHORTPROTOCOL_Send(SHORTPROTOCOL_Instance* inst,
18 uint8_t* data, uint32_t length){
19 /* Command Length without Zero Termination */
20 /* This Function generates a Shortprotocol Packet from a Command */
21
22 /* If the last Packet is not yet send, cant accept a new Command */
23 if(inst->newCommand == SHORTPROTOCOL_NOT_AVAILABLE){
24 return SHORTPROTOCOL_NOT_AVAILABLE;
25 }
26
27 /* Check that the Command will fit in the Packet Buffer */
28 if(length > SHORTPROTOCOL_MAXIMUM_COMMAND_LENGTH){
29 length = SHORTPROTOCOL_MAXIMUM_COMMAND_LENGTH;
30 }
31
32 /* Check that the Command is not zero length */
33 if(length == 0){
34 return SHORTPROTOCOL_ERROR;
35 }
36
37 /* First Byte of Packet is Always the Same */
38 inst->command_buffer.data[SHORTPROTOCOL_BEGIN_OFFSET] = SHORTPROTOCOL_BEGIN;
39
40 /* Second and third Byte is the Length of the Command */
41 inst->command_buffer.data[SHORTPROTOCOL_LENGTH_LSB_OFFSET] =
42 (length & SHORTPROTOCOL_FIRST_BYTE_MASK);
43
44 inst->command_buffer.data[SHORTPROTOCOL_LENGTH_MSB_OFFSET] =
45 ((length & SHORTPROTOCOL_SECOND_BYTE_MASK)
46 >> SHORTPROTOCOL_BYTE_LENGTH);
47
48 /* The Command begins on the fourth byte */
49 for(uint32_t i=0; i<length; i++){
50 inst->command_buffer.data[i + SHORTPROTOCOL_COMMAND_OFFSET] = data[i];
51 }
52
53 /* Calculate the CRC of the Packet */
54 uint32_t crc_temp;
55 crc_temp = CRC_Calculate(inst->command_buffer.data,
56 length + SHORTPROTOCOL_OVERHEAD_WITHOUT_CRC_BYTES);
57
58 /* The last two Bytes are the CRC */
59 inst->command_buffer.data[SHORTPROTOCOL_OVERHEAD_WITHOUT_CRC_BYTES +
60 length + SHORTPROTOCOL_CRC_LSB_OFFSET] =
61 crc_temp & SHORTPROTOCOL_FIRST_BYTE_MASK;
62
63 inst->command_buffer.data[SHORTPROTOCOL_OVERHEAD_WITHOUT_CRC_BYTES +
64 length + SHORTPROTOCOL_CRC_MSB_OFFSET] =
65 (crc_temp & SHORTPROTOCOL_SECOND_BYTE_MASK)
66 >> SHORTPROTOCOL_BYTE_LENGTH;
67
68 /* Calculate the Length of the Full Packet */
69 inst->command_buffer.length = length + SHORTPROTOCOL_OVERHEAD_BYTES;
70
71 /* Tell the Update Function a Packet is ready */
72 inst->writeCounter = 0;
73 inst->newCommand = SHORTPROTOCOL_NOT_AVAILABLE;
74
75 return SHORTPROTOCOL_SUCCESS;
76}

78void SHORTPROTOCOL_Update(SHORTPROTOCOL_Instance* inst){
79 /*
80 * If a New Command is not available
81 * (therefore the current one is not completed)
82 */
83 if(inst->newCommand == SHORTPROTOCOL_NOT_AVAILABLE){
84 /* Counter to control the length of this package being sent */
85 uint32_t package_length = 0;
86 while(package_length < inst->maximumPackageLength){
87
88 inst->writeByte(inst->command_buffer.data[inst->writeCounter]);
89
90 while(inst->writeAvailable() == SHORTPROTOCOL_NOT_AVAILABLE);
91
92 if(inst->writeCounter == inst->command_buffer.length - 1){
93 /* Command was completly written, a new Command is writable */
94 inst->newCommand = SHORTPROTOCOL_AVAILABLE;
95 break;
96 }
97
98 inst->writeCounter += 1;
99 package_length += 1;
100 }
101 }
102}
