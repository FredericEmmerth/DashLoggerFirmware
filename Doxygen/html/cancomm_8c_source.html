17void CANCOMM_ReadMessages(cancomm_message* message_list,
18 uint32_t message_list_len,
19 cancomm_interface* interface_list,
20 uint8_t interface_list_len){
21
22 /* Local Variables where the Frame Data can be temporarily saved */
23 uint8_t message_recieved = 0;
24 uint32_t temp_id;
25 uint8_t temp_length;
26 uint8_t temp_data[CANCOMM_MAXIMUM_DATA_LENGTH];
27 CANFD_MSG_RX_ATTRIBUTE temp_msgAttr;
28 uint32_t temp_timestamp;
29
30 /* Loop over every Interface and get the recieved Messages from the FIFOs*/
31 for(uint8_t current_interface = 0;
32 current_interface < interface_list_len;
33 current_interface ++)
34 {
35
36 /* Endless Loop, broken internally when there is no data in the fifo */
37 while(1){
38
39 /* Try to get a Message from FIFO */
40 message_recieved = interface_list[current_interface].MessageReceive(
41 &temp_id,
42 &temp_length,
43 temp_data,
44 &temp_timestamp,
45 interface_list[current_interface].receiveFifo,
46 &temp_msgAttr);
47
48 /* If there was a Message in the FIFO interpret it */
49 if(message_recieved){
50 /* See if the Message is in cancomm_message_list */
51 for(uint32_t i=0; i<message_list_len; i++){
52 /* If ID and Interface match, copy the data into the message_list*/
53 if((message_list[i].id == temp_id)&&
54 (message_list[i].interface_number ==
55 interface_list[current_interface].number))
56 {
57 for(uint8_t j=0; j<CANCOMM_MAXIMUM_DATA_LENGTH; j++){
58 message_list[i].data[j] = temp_data[j];
59 }
60 /* Break the Loop after finding the Message */
61 break;
62 }
63 }
64 /* If there was no Message in the FIFO break Loop */
65 }else{
66 break;
67 }
68 }

70 }
