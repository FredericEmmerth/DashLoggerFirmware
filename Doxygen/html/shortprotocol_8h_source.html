11/**
12 * \defgroup SHORTPROTOCOL SHORTPROTOCOL
13 * This Module handles packing the Commands generated by the \ref SIGNAL Module
14 * and sending them using the Shortprotocol defined in the Displays
15 * (EA uniTFTs035-ATC) Datasheet. To not block the Main Loop for too long, the
16 * Packet is split into smaller Parts defined by
17 * SHORTPROTOCOL_Instance.maximumPackageLength. The Module is written in a
18 * general Fashion and can be adopted to use SPI, for this the Callback Functions
19 * in SHORTPROTOCOL_Instance have to be redefined for SPI.
20 *
21 * \addtogroup SHORTPROTOCOL
22 * \{
23 */
24

127/**
128 * @brief Configuration Struct for \ref SHORTPROTOCOL passed to the
129 * \ref SHORTPROTOCOL Functions
130 *
131 * The Callback functions have to all be defined as described in thier
132 * Documentaion, also a maximumPackageLength has to be defined.
133 */
134typedef struct{
135 /**
136 * @brief Callback to read one Byte from the Serial Interface
137 */
138 uint8_t (*readByte)(void);
139
140 /**
141 * @brief Callback to check if a Byte is available to be read.
142 * If this returns SHORTPROTOCOL_AVAILABLE at least one Byte has to be
143 * readable from readByte.
144 */
145 SHORTPROTOCOL_status (*readAvailable)(void);
146
147 /**
148 * @brief Callback to write one Byte to the Serial Interface
149 */
150 void (*writeByte)(uint8_t);
151
152 /**
153 * @brief Callbakc to check if a Byte is writeable to the Serial Interface
154 * If this returns SHORTPROTOCOL_AVAILABLE at leat one Byte has to be
155 * writeable to write Byte.
156 */
157 SHORTPROTOCOL_status (*writeAvailable)(void);
158
159 /**
160 * @brief The maximum Package Length
161 * Packets longer than this will be split into Packets with the maximum Size
162 * of maximumPackageLength.
163 */
164 uint32_t maximumPackageLength;
165
166 /**
167 * @brief A Buffer where the Command wich is currently sent resides.

169 SHORTPROTOCOL_string command_buffer;
170
171 /**

174 */
175 SHORTPROTOCOL_status newCommand;
176
177 /**

181 uint32_t writeCounter;

183}SHORTPROTOCOL_Instance;

185/**
186 * @brief Function to try to send a new Command using the SHORTPROTOCOL
187 *
188 * \param inst SHORTPROTOCOL_Instance
189 * \param data The Data to be sent using the Shortprotocol
190 * \param length The Length of the Payload (Data) to be sent
191 * (without NULL termination)
192 *
193 * @return SHORTPROTOCOL_NOT_AVAILABLE when the last Command was not yet sent,
194 * SHORTPROTOCOL_ERROR when the Command has a length of zero,
195 * SHORTPROTOCOL_SUCCESS when the Command was successfully queued to be sent.
196 */
197SHORTPROTOCOL_status SHORTPROTOCOL_Send(SHORTPROTOCOL_Instance* inst,
198 uint8_t* data, uint32_t length);

200/**
